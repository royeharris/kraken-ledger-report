    <script>
    document.getElementById('kraken-ledger-form').addEventListener('submit', async (event) => {
        event.preventDefault();

        // Get user input from the form
        const startDate = document.getElementById('start-date').value;
        const endDate = document.getElementById('end-date').value;
        const walletAddresses = document.getElementById('wallet-addresses').value.split(',');

        // Your Kraken API key and private key
        const apiKey = 'your_api_key';
        const privateKey = 'your_private_key';

        // Send API request to Kraken with axios
        try {
            const response = await axios({
                method: 'POST',
                url: 'https://api.kraken.com/0/private/GetLedgers',
                headers: {
                    'API-Key': apiKey
                },
                data: {
                    start: startDate,
                    end: endDate,
                    // Add other parameters as needed
                }
            });

            // Handle the response and update the table with results
            const ledgerData = response.data.result;
            const ledgerTableBody = document.getElementById('ledger-data');
            ledgerTableBody.innerHTML = '';

            for (const entry of ledgerData) {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${entry.recipient}</td>
                    <td>${entry.asset}</td>
                    <td>${entry.gbpValue}</td>
                    <td>${entry.usdtEquivalent}</td>
                    <td>${entry.time}</td>
                `;
                ledgerTableBody.appendChild(row);
            }
        } catch (error) {
            console.error(error);
            // Handle any errors here
        }
    });
    </script>
